            mimeType="text/plain"

üîÑ Protocol Communication
// MCP Client-Server Communication
class MCPProtocol {
  // Initialize connection
  async initialize() {
    const request = {
      method: "initialize",
      params: {
        protocolVersion: "2024-11-05",
        capabilities: {
          tools: {}, resources: {}, prompts: {}
        },
        clientInfo: {
          name: "enterprise-client",
          version: "1.0.0"
        }
      }
    };
    
    const response = await this.sendRequest(request);
    await this.sendNotification({ 
      method: "initialized" 
    });
  }

  // Execute tool with progress reporting
  async callTool(name, arguments) {
    return await this.sendRequest({
      method: "tools/call",
      params: { name, arguments }
    });
  }
}

// Tool Implementation with External API
server.setRequestHandler(CallToolRequestSchema, 
  async (request) => {
    if (request.params.name === "execute_sql_query") {
      const { query, database } = request.params.arguments;
      
      try {
        // Execute database query via API
        const response = await fetch(`/api/databases/${database}/query`, {
          method: "POST",
          headers: { 
            "Content-Type": "application/json",
            "Authorization": `Bearer ${process.env.DB_TOKEN}`
          },
          body: JSON.stringify({ query })
        });
        
        const data = await response.json();
        
        return {
          content: [{
            type: "text",
            text: `Query executed successfully. 
Results: ${JSON.stringify(data, null, 2)}`
          }]
        };
      } catch (error) {
        return {
          content: [{
            type: "text",
            text: `Error: ${error.message}`
          }],
          isError: true
        };
      }
    }
  });

// Connect LLM to MCP Server
import { Client } from "@modelcontextprotocol/sdk/client";
import { StdioClientTransport } from "@modelcontextprotocol/sdk/client/stdio";

class LLMWithMCP {
  constructor() {
    this.mcpClient = new Client({
      name: "enterprise-llm",
      version: "1.0.0"
    }, {
      capabilities: {
        tools: {}
      }
    });
  }

  async connectToMCPServer(serverPath) {
    const transport = new StdioClientTransport({
      command: "node",
      args: [serverPath]
    });

    await this.mcpClient.connect(transport);
    
    // Get available tools
    const tools = await this.mcpClient.listTools();
    console.log("Available tools:", tools);
    
    return tools;
  }

  async executeWithTools(prompt) {
    // LLM processes prompt and decides to use tools
    const result = await this.mcpClient.callTool({
      name: "execute_sql_query",
      arguments: {
        query: "SELECT * FROM customers LIMIT 10",
        database: "production"
      }
    });
    
    return result;
  }
}
